OPTION     CASEMAP:NONE                   ;чувствительность к регистру

IncludeLib C:\masm32\lib\kernel32.lib

.CONST  
    Arr DB 11, 2, 3, 4, 5, 0, -7, 8, 9, 10
    SizeArr DB 10

.CODE 
Start:     
    ;Определить, есть ли отрицательные числа в массиве из 10 элементов
    mov ebx, OFFSET Arr
    xor esi, esi 
    xor eax, eax

    movzx ecx, BYTE PTR [SizeArr]
    For_start:
        mov dl, BYTE PTR [ebx + esi]
        
        test dl, dl
        jge Inc_for

        inc eax ; есть отрицательные EAX = 1, нет => 0
        jmp Exit

    Inc_for:
        inc esi
        loop For_start
    
    Exit:
        Invoke ExitProcess, 0              ;Выход с кодом 0 
End Start
