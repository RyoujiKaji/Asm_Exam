OPTION     CASEMAP:NONE                   ;чувствительность к регистру 

Include    C:\masm32\include\kernel32.inc
IncludeLib C:\masm32\lib\kernel32.lib

.CONST  
    Arr DB -11, 2, 3, -4, 5, 0
    SizeArr DB 6

.CODE 
Start:     
    ;Массив целых чисел заканчивается на 0. 
    ;найти сумму положительных чисел массива
    mov ebx, OFFSET Arr
    xor esi, esi 
    xor edx, edx

    movzx ecx, BYTE PTR [SizeArr]
    For_start:
        movsx eax, BYTE PTR [ebx + esi]
        
        test eax, eax
        jz Exit
        jl Ind_inc

        add edx, eax

    Ind_inc:
        inc esi
        loop For_start

    Exit:
        Invoke ExitProcess, 0              ;Выход с кодом 0 
End Start
