.MODEL     flat, stdcall
OPTION     CASEMAP:NONE                   ;чувствительность к регистру 

Include    C:\masm32\include\kernel32.inc
IncludeLib C:\masm32\lib\kernel32.lib

.CONST  
    Arr DB 11, 2, 3, 4, 5, 0, 7, 8, 9, 10
    SizeArr DB 10

.DATA 
    Square DD 10 DUP (0)
    Cube DD 10 DUP (0)

.CODE 
Start:     
    ;Создать массив квадратов и кубов для диапазона чисел 20-30 (2 массива)
    mov ebx, OFFSET Arr
    mov esi, OFFSET Square
    mov edi, OFFSET Cube

    movzx ecx, BYTE PTR [SizeArr]
    For_start:
        dec ecx
        movsx eax, BYTE PTR [ebx + ecx]
        mov edx, eax
        
        imul eax, eax
        mov [esi + ecx*4], eax

        imul eax, edx
        mov [edi + ecx*4], eax
        
        test ecx, ecx
        jnz For_start

    Invoke ExitProcess, 0              ;Выход с кодом 0 
End Start
