Start:     
    ;Вывести значения и количество четных чисел в массиве из 10 элементов
    mov ebx, OFFSET Arr
    xor esi, esi 
    xor edx, edx

    movzx ecx, BYTE PTR [SizeArr]
    For_start:
        movsx eax, BYTE PTR [ebx + esi]
        
        test eax, 1
        jnz Ind_inc

        inc edx
        ;вывод четного числа

    Ind_inc:
        inc esi
        loop For_start

    Invoke ExitProcess, 0              ;Выход с кодом 0 
End Start
