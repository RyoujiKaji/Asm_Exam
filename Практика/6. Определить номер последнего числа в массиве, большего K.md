.CODE 
Start:     
    ;Определить номер последнего числа в массиве, большего K
    mov ebx, OFFSET Arr
    mov eax, -1 ; нет числа больше
    mov di, K

    movzx ecx, BYTE PTR [SizeArr]
    For_start:
        dec ecx
        mov dx, [ebx + ecx*2]
        
        cmp dx, di
        jle Inc_for

        mov eax, ecx
        jmp Exit
        
    Inc_for:
        test ecx, ecx
        jnz For_start
    
    Exit:
        Invoke ExitProcess, 0              ;Выход с кодом 0 
End Start
